393. UTF-8 Validation

位运算：

class Solution {
public:
    bool validUtf8(vector<int>& data) {
        int n = data.size(), status = 0;
        for(int i = 0; i < n; i++) {
            int mask = 1 << 7, prefix = data[i], count = 0;
            while(mask & prefix) {
                count++;
                mask = mask >> 1;
            }
            if(count > 4) return false;
            if(count > 1) {
                if(status > 0) return false;
                status += count - 1;
            } else if(count == 1) {
                status--;
            }
            
        }
        return status == 0;
    }
};

Notes:

  1. “1”的数量大于4时，直接return false;
  2. count > 1时’status > 0’意味着byte数与“1”的数量不符，return false;
  3. 保证最后加减相抵消，即可保证算法正确性

290. Word Pattern

踩坑点：

  1. c++ 下如何split：

​ 对于 " "来说，可以使用sstream来进行分割

stringstream ss(s);
istream_iterator<string> begin(ss), end;
vector<string> s_arr(begin, end);
  1. 相互映射，不止string->char，还得char->string

源代码：

class Solution {
public:
    bool wordPattern(string pattern, string s) {
        stringstream ss(s);
        istream_iterator<string> begin(ss), end;
        vector<string> s_arr(begin, end);
        if(pattern.size() != s_arr.size()) return false;
        int n = pattern.size();
        unordered_map<char, string> m;
        unordered_map<string, char> h;
        for(int i = 0; i < n; i++) {
            if(m.count(pattern[i]) == 0 && h.count(s_arr[i]) == 0) {
                m[pattern[i]] = s_arr[i];
                h[s_arr[i]] = pattern[i];
            } else if(m[pattern[i]] != s_arr[i] || h[s_arr[i]] != pattern[i]) {
                return false;
            }
        }
        return  true;
    }
};

732. My Calendar III

核心思想：扫描线段

用一个map记录多个线段的起点和终点，起点++，终点–，最后从小到大扫描map，记录最大的重叠部分

class MyCalendarThree {
private:
    map<int, int> m;
public:
    MyCalendarThree() {
        
    }
    
    int book(int start, int end) {
        m[start]++;
        m[end]--;
        int ans = 0, lad = 0;
        for(auto &[_, num] : m) {
            lad += num;
            ans = max(ans, lad);
        }
        return ans;
    }
};

 * MyCalendarThree* obj = new MyCalendarThree();
 * int param_1 = obj->book(start,end);

218. The Skyline Problem

踩坑记录

  1. 要处理单横坐标点上存在多个建筑的情况
  2. 用于存储的二维数组，横坐标要从小到大排序，而高度要从大到小排序才能保证天际线不会因为单坐标点上出现多个建筑导致结果“冗余”
  3. 由于单坐标点存在多个建筑，因此不能再维护一个单调数组（’c++中可用set实现）了，而是应该使用map来记录每个横坐标点上的建筑数量

代码：

class Solution {
public:
    vector<vector<int>> getSkyline(vector<vector<int>>& buildings) {
        vector<pair<int, int>> key_points;
        
        for(auto building : buildings) {
            int left = building[0], right = building[1], height = building[2];
            key_points.push_back({left, height});
            key_points.push_back({right, -height});
        }
        sort(key_points.begin(), key_points.end(), [](auto a, auto b) {
            if(a.first == b.first) return a.second > b.second;
            else return a.first < b.first;
        });
        map<int, int, greater<int>> points;
        vector<vector<int>> ans;
        int max_height = 0;
        for(auto &[index, height] : key_points) {
            if(height > 0) points[height]++;
            else points[-height]--;
            if(points[abs(height)] == 0) points.erase(abs(height));
            int top = points.begin()->first;
            if(top != max_height)ans.push_back({index, top});
            max_height = top;
        }
        return ans;
    }
};

300. Longest Increasing Subsequence

✔️ Solution 2: Greedy with Binary Search

  • Let’s construct the idea from following example.

  • Consider the example

    nums = [2, 6, 8, 3, 4, 5, 1]

    , let’s try to build the increasing subsequences starting with an empty one:

    sub1 = []

    .

    1. Let pick the first element, sub1 = [2].
    2. 6 is greater than previous number, sub1 = [2, 6]
    3. 8 is greater than previous number, sub1 = [2, 6, 8]
    4. 3 is less than previous number, we can’t extend the subsequence sub1, but we must keep 3 because in the future there may have the longest subsequence start with [2, 3], sub1 = [2, 6, 8], sub2 = [2, 3].
    5. With 4, we can’t extend sub1, but we can extend sub2, so sub1 = [2, 6, 8], sub2 = [2, 3, 4].
    6. With 5, we can’t extend sub1, but we can extend sub2, so sub1 = [2, 6, 8], sub2 = [2, 3, 4, 5].
    7. With 1, we can’t extend neither sub1 nor sub2, but we need to keep 1, so sub1 = [2, 6, 8], sub2 = [2, 3, 4, 5], sub3 = [1].
    8. Finally, length of longest increase subsequence = len(sub2) = 4.

  • In the above steps, we need to keep different sub arrays (sub1, sub2…, subk) which causes poor performance. But we notice that we can just keep one sub array, when new number x is not greater than the last element of the subsequence sub, we do binary search to find the smallest element >= x in sub, and replace with number x.

  • Let’s run that example

    nums = [2, 6, 8, 3, 4, 5, 1]

    again:

    1. Let pick the first element, sub = [2].
    2. 6 is greater than previous number, sub = [2, 6]
    3. 8 is greater than previous number, sub = [2, 6, 8]
    4. 3 is less than previous number, so we can’t extend the subsequence sub. We need to find the smallest number >= 3 in sub, it’s 6. Then we overwrite it, now sub = [2, 3, 8].
    5. 4 is less than previous number, so we can’t extend the subsequence sub. We overwrite 8 by 4, so sub = [2, 3, 4].
    6. 5 is greater than previous number, sub = [2, 3, 4, 5].
    7. 1 is less than previous number, so we can’t extend the subsequence sub. We overwrite 2 by 1, so sub = [1, 3, 4, 5].
    8. Finally, length of longest increase subsequence = len(sub) = 4.

38. Count and Say

踩坑记录：

在for循环中进行初始化和尾处理，反而可以使代码简化！！

class Solution {
public:
    string countAndSay(int n) {
        if(n == 1) return "1";
        string said = countAndSay(n - 1);
        int count = 0, m = said.size();
        string ans;
        for(int i = 0; i < m; i++)  {
           if(i && said[i] != said[i - 1]) {
               ans += to_string(count) + to_string(said[i - 1] - '0');
               count = 1;
           } else {
               count++;
           }
           if(i == m - 1) ans += to_string(count) + to_string(said[i] - '0');
        }
        return ans;
    }
};

if(i && ...) 以及if(i == m - 1)，可以使代码变得更加简单，不用在循环外部进行特判！

16. 3Sum Closest

核心思想：双指针

双指针本质是二分查找的一种，一般用于查找两个数的和，对于有序数列，两个指针从两端出发，如果当前两个的和大于目标值，则右指针左移；小于则左指针右移

代码：

class Solution {
public:
    int threeSumClosest(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int ans = 1e5, n = nums.size();
        for(int i = 0; i < n; i++) {
            int left = i + 1, right = n - 1;
            while(left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                ans = (abs(sum - target) < abs(ans - target) ? sum : ans);
                if(sum > target) right--;
                else if (sum < target) left++;
                else goto ret;
            }
        }
        ret:
        return ans;
    }
};

886. Possible Bipartition

二分图问题：

踩坑日志：

  1. 将奇怪的dislikes格式转换成接邻矩阵
  2. 二分图可能有多个不连通的图构成，因此是多源的，需要for(int i = 1; i <= n; i++)遍历一遍node
class Solution {
public:
    bool possibleBipartition(int n, vector<vector<int>>& dislikes) {
        vector<vector<int>> edges(n + 1, vector<int>());
        vector<int> label(n + 1, 0);
        queue<int> buffer;
        if(dislikes.empty()) return true;
        for(auto p : dislikes) {
            if(p.empty()) continue;
            edges[p[0]].push_back(p[1]);
            edges[p[1]].push_back(p[0]);
        }

        for(int i = 1; i <= n; i++) {
            if(label[i] == 0) buffer.push(i);
            else continue;

            label[i] = 1;
            while(!buffer.empty()) {
                auto p = buffer.front();
                buffer.pop();
                if(label[p] == 1) {
                    for(auto edge : edges[p]) {
                        if(label[edge] == 1) return false;
                        else if(label[edge] == 0) {
                            label[edge] = 2;
                            buffer.push(edge);
                        }
                       
                        
                    }
                } else if(label[p] == 2) {
                    for(auto edge : edges[p]) {
                        if(label[edge] == 2) return false;
                        else if(label[edge] == 0) {
                            buffer.push(edge);
                            label[edge] = 1;
                        }
                    }
                }
            }
        }

        return true;
    }
};

19. Remove Nth Node From End of List

经验教训：

链表操作最少需要两个指针！prev和core，其他的只可能比这个多，比如快慢指针需要三个！

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode *guard = new ListNode(0, head);
        ListNode *prev = guard, *cur = guard, *nxt = guard;
        for(int i = 0; i < n; i++) nxt = nxt->next;
        while(nxt != nullptr) {
            nxt = nxt->next;
            prev = cur;
            cur = cur->next;
        }
        prev->next = cur->next;
        return guard->next;
    }
};

981. Time Based Key-Value Store

踩坑记录：

  1. 双重map的插入方法: m[key].insert({timestamp, value});，采用insert方法直接插入pair类型
  2. dead_signal: 在查找时，需要判断第一层map的key是否存在
class TimeMap {
private:
    unordered_map<string, map<int, string, less<int>>> m;
public:
    TimeMap() {
        
    }
    
    void set(string key, string value, int timestamp) {
        m[key].insert({timestamp, value});
    }
    
    string get(string key, int timestamp) {
        if(m.count(key) == 0) return "";
        auto p = m[key].upper_bound(timestamp);
        p--;
        if(p->first <= timestamp) return p->second;
        else return "";
    }
};
