割绳子问题

问题：

总共有n条长度不等绳子，可以任意切割，不能拼接。要求切割后得到m条长度相等的绳子，求问得到的m条绳子长度L的最大值

输入 ： 绳子条数n；表示绳子长度的n维数组l，要求输出的绳子长度m

输出： 长度最大值L

思路：

先考虑一种简化的情形，假设每一根绳子只能使用一次（即使绳子很长，可以切割出多条长度为L的小段，也只使用其中一段），那么这个问题就退化为一个简单的排序问题，直接找出其中第m长的绳子，其长度就是L的最大值。这种情形很简单。

再考虑一种稍微复杂一些的情形：假设每一根绳子最多只能使用两次，那么可以对l[0], l[1], …, l[n-1], l[0]/2, l[1]/2, …, l[n-1]/2 这2n个数排序，取其中第m大的数，其值就是L的最大值。

举个栗子，假设有n=4根绳，长度为a>b>c>d，又有b>a/2>c，要取出m=3条长度相等的绳。则对4*2=8个数排序后前三位是a,b,a/2. 那么长度L的最大值就是a/2；前三位中最长的绳出现了两次（a，a/2），第二长的绳出现了一次，而切割出的3个片段中最长的绳可以切出两个，第二次的绳可以切出一个，也就是说，每一根绳分出的片段数和它在排序后的数组的前m位中出现的次数是对应的。

如果把上面的条件改为a/2>b，则L的最大值就是b，同样是a分出两段，b分出一段。条件改为a/2<c，则L最大值为c，三根绳a, b, c各分出一段。

回到原来的问题，因为不限制每根绳的使用次数，所以一根绳最多使用m次（极端情况a/m>b），因此对l[0], l[1], …, l[n], l[0]/2, l[1]/2, …, l[n]/2, l[0]/3, …, l[n-1]/m共m*n个数排序，再取其第m大的元素既是L的最大值。这种情况下，“每一根绳分出的片段数和它在排序后的数组的前m位中出现的次数相等”这一条件依然能得到满足。

代码实现：

#include <bits/stdc++.h>
using namespace std;



const int N = 0xffff;
int arr[N], m, n;


int max_len() {
   priority_queue<int> len;
   for(int i = 0; i < n; i++) {
      for(int j = 1; j <= m; j++) {
         len.push(arr[i] / j);
      }
   }
   int ans = 0;
   for(int i = 0; i < m; i++) {
      ans = len.top();
      len.pop();
   }
   return ans;
}


int main() {
   cin >> n >> m;
   for(int i = 0; i < n; i++) {
      scanf("%d", &arr[i]);
   }
   cout << max_len() << endl;
   return 0;
}

vector<int>getSa(string s)
{
    int n=s.size(),m=130;//
    vector<int>sa(n+1,0),rk(n+1,0),pos(n+1,0),cnt(m,0);
    auto base_sort=[&](){
        fill(cnt.begin(),cnt.begin()+m+1,0);
        for(int i=1;i<=n;++i)cnt[rk[i]]++;
        for(int i=2;i<m;++i)cnt[i]+=cnt[i-1];
        for(int i=n;i>=1;--i)sa[cnt[rk[pos[i]]]--]=pos[i];//此时第一位的次序依旧不变，此时还是用的前一次的rank，
        //不同的是pos[i]的改变，对于rk[pos[i]]==rk[pos[j]]来说，如果你i<j，那么你的会慢一点遍历到，对应的sa值也就会小
    };
    auto check=[&](int x,int y,int k){
         if(pos[x]!=pos[y])
         {
            return false;
         }
         int r1=x+k>n?-1:pos[x+k];
         int r2=y+k>n?-1:pos[y+k];
         return r1==r2;
    };
    for(int i=1;i<=n;++i)rk[i]=s[i-1],pos[i]=i;
    base_sort();
    for(int k=1;k<=n;k<<=1)
    {
        int rank=0;for(int i=n-k+1;i<=n;++i)pos[++rank]=i;
        for(int i=1;i<=n;++i)if(sa[i]>k)pos[++rank]=sa[i]-k;
        base_sort();
        swap(rk,pos);
        rank=1;rk[sa[1]]=rank;
        //这个时候的pos数组是rk
        for(int i=2;i<=n;++i){if(!check(sa[i],sa[i-1],k))++rank;rk[sa[i]]=rank;}
        if(rank==n)break;//完全排好序了
        m=rank;
    }
    return sa;
}

    for(int i=1;i<=n;++i)rk[sa[i]]=i;
    int h=0;
    lcp[1]=0;
    for(int i=1;i<=n;++i)
    {
        int k=sa[rk[i]-1];

        for(;k+h<=n&&i+h<=n;++h)
        {
            if(s[i+h-1]!=s[k+h-1])break;
        }
        lcp[rk[i]]=h;
    }
    return lcp;
}

421. Number of Good Paths

核心：并查集

关键要把并查集UnionFind写对，注意一些优化方法，注意是p[x] = find(p[x])而不是p[x]=find(x)

class Solution {
private:
    struct UnionFind {
        vector<int> p, size;
        UnionFind(int n) {
            size.resize(n, 1);
            for(int i = 0; i < n; i++) p.push_back(i);
        }
        
        inline int find(int x) {
            return x == p[x] ? x : (p[x] = find(p[x]));
        }
        
        void join(int x, int y) {
            int fx = find(x), fy = find(y);
            if(fx == fy) return;
            if(size[fx] > size[fy]) swap(fx, fy);
            p[fx] = fy;
            size[fx] += size[fy];
        }
    };
public:
    int numberOfGoodPaths(vector<int>& vals, vector<vector<int>>& edges) {
        int n = vals.size();
        UnionFind uf(n);
        unordered_map<int, vector<int>> g;
        map<int, vector<int>> o_node;
        int ans = 0;
        for(auto &edge : edges) {
            int u = edge[0], v = edge[1];
            g[u].push_back(v);
            g[v].push_back(u);
        }
        
        for(int i = 0; i < n; i++) {
            o_node[vals[i]].push_back(i);
        }
        
        for(auto &[_, nodes] : o_node) {
            unordered_map<int, int> count;
            for(auto &node : nodes) {
                for(auto &adj : g[node]) {
                    if(vals[adj] <= vals[node]) uf.join(adj, node);
                    //cout << "1:" << adj << " " << node << endl;
                }
            }
            
            for(auto &node : nodes) {
                count[uf.find(node)]++;
            }
            
            for(auto &[_, num] : count) {
                ans += num * (num - 1) / 2;
            }
            //cout << ans << endl;
        }
        return ans + n;
    }
};

2420. Find All Good Indices

Notes:

  1. 主题思想单调队列没有任何问题，关键是如何实现单调队列。通过单一从左到右遍历数组和容易出现一些情况无法考虑到的情况（几十个WA的教训），因此考虑从数组两端进行两次遍历，分别统计出上升和下降的长度
  2. 当increase[i + 1]和decrease[i - 1]都大于k时，则当前的i满足题目要求
class Solution {
public:
    vector<int> goodIndices(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> increase(n, 1), decrease(n, 1), ans;
        
        for(int i = 1; i < n; i++) {
            if(nums[i] <= nums[i - 1]) decrease[i] += decrease[i - 1];
        }
        
        for(int i = n - 2; i >= 0; i--) {
            if(nums[i] <= nums[i + 1]) increase[i] += increase[i + 1];
        }
        
        for(int i = k; i < n - k; i++) {
            if(increase[i + 1] >= k && decrease[i - 1] >= k) ans.push_back(i);
        }
        return ans;
    }
};

2423. Remove Letter To Equalize Frequency

踩坑记录：

统计维护最高频率max_freq和次高频率max_freq - 1，满足下列三种情况返回true

  1. 最高频率的字母一共有max_count=1个，意味着剩下的全是max_freq - 1频率的字母
  2. 最高频率为1，其他字母可以直接删除掉
  3. 最低频率的字母有且只有一个
class Solution {
public:
    bool equalFrequency(string word) {
        unordered_map<char, int> m;
        int max_freq = 0, max_count = 0, min_count = 0;
        for(auto c : word) {
            m[c]++;
            max_freq = max(max_freq, m[c]);
        }
        for(auto &[_, freq] : m) {
            if(freq == max_freq) max_count++;
            else if(freq == max_freq - 1) min_count++;
            else return false;
        }
        
        return max_count == 1 || max_freq == 1 || min_count == 1;
    }
};

class Solution {
public:
    int numberOfWeakCharacters(vector<vector<int>>& properties) {
        sort(properties.begin(), properties.end(), [](vector<int> &a, vector<int> &b) {
            if(a[0] != b[0]) return a[0] > b[0];
            else return a[1] < b[1];
        });
        
        int ans = 0, max_def = INT_MIN;
        for(auto p : properties) {
            if(max_def > p[1]) ans++;
            else max_def = p[1];
        }
        return ans;
    }
};

核心思想：

attack属性从大到小排列，defend属性从小到大排列，从而确保当遍历到一个auto p时，当max_def比当前p的defend值大时，attack也能保证比defend大。

2007. Find Original Array From Doubled Array

class Solution {
public:
    vector<int> findOriginalArray(vector<int>& changed) {
        if(changed.size() % 2 == 1) return {};
        unordered_map<int, int> m;
        vector<int> ans;
        for(auto num : changed) {
            m[num]++;
        }
        sort(changed.begin(), changed.end());
        for(auto num : changed) {
            if(m[num] == 0) continue;
            if(m[num * 2] == 0) return {};
            ans.push_back(num);
            m[num]--;
            m[num * 2]--;
        }
        return ans;
    }
};

Notes

  1. 为什么用map而不是set？

    因为要对未来遍历到的元素进行“删除”操作，如果用set会导致越界

  2. 为什么要对changed排序后遍历，而不是用map统计完changed后直接遍历map，如：

1531. String Compression II

思路分析：

状态转移方程：

dp[i][j]表示第i个字符在删掉j个字符的时候，压缩字符串的长度

如果选择直接删除当前字符，那么dp[i][j] = dp[i + 1][j - 1]

如果不删除当前字符，则从当前往后进行遍历，将与当前字符不同的字符则删除，直到用完删除次数为止：

inline int getcount(int count) {
       return (count == 1 ? 1 : (count < 10 ? 2 : (count < 100 ? 3 : 4)));
}

...

int del_c = j, count = 0;
for(int k = i; k < n && del_c >= 0; k++) {
    if(s[i] == s[k]) {
        count++;
        //这里确保只有当s[i] != s[k + 1]时，才会取到最小值，在取到最小值前的主要目的为更新count
        dp[i][j] = min(dp[i][j], dp[k + 1][del_c] + getcount(count));
    } else {
        del_c--;
    }
}

边界条件：

dp[n][x]时代表一个字符都没选，因此初始长度为0

dp[x][0]表示删了0个字符，因此字符串长度为原长n，但可以用一个随意的最大值替代，如INT_MAX

代码实现：

class Solution {
private:
   int dp[101][101];
   inline int getcount(int count) {
       return (count == 1 ? 1 : (count < 10 ? 2 : (count < 100 ? 3 : 4)));
   }
public:
    int getLengthOfOptimalCompression(string s, int del) {
        int n = s.size();
        for(int i = n; i >= 0; i--) {
            for(int j = 0; j <= del; j++) {
                if(i == n) {
                    dp[i][j] = 0;
                    continue;
                }
                dp[i][j] = (j == 0 ? INT_MAX : dp[i + 1][j - 1]);
                int del_c = j, count = 0;
                for(int k = i; k < n && del_c >= 0; k++) {
                    if(s[i] == s[k]) {
                        count++;
                        dp[i][j] = min(dp[i][j], dp[k + 1][del_c] + getcount(count));
                    } else {
                        del_c--;
                    }
                }
            }
        }
        return dp[0][del];
    }
};

114. Flatten Binary Tree to Linked List

Given the root of a binary tree, flatten the tree into a “linked list”:

  • The “linked list” should use the same TreeNode class where the right child pointer points to the next node in the list and the left child pointer is always null.
  • The “linked list” should be in the same order as a pre-order traversal of the binary tree.

Example 1:

Input: root = [1,2,5,3,4,null,6]
Output: [1,null,2,null,3,null,4,null,5,null,6]

Code:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    TreeNode* dfs(TreeNode* root) {
        if(!root->left && !root->right) return root;
        TreeNode *tail, *next = root->right;
        
        if(root->left != nullptr) {
            tail = dfs(root->left);
            root->right = root->left;
            root->left = nullptr;
            tail->right = next;
        }
        if(root->right != nullptr) tail = dfs(root->right);
        return tail;
    }
public:
    void flatten(TreeNode* root) {
        if(root == nullptr) return;
        dfs(root);
        return;
    }
};

Solution:

Flatten the left subtree, then insert it between the root and the right child, finally go to the right tree and repeat the process before.
