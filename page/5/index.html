2420. Find All Good Indices

Notes:

  1. 主题思想单调队列没有任何问题，关键是如何实现单调队列。通过单一从左到右遍历数组和容易出现一些情况无法考虑到的情况（几十个WA的教训），因此考虑从数组两端进行两次遍历，分别统计出上升和下降的长度
  2. 当increase[i + 1]和decrease[i - 1]都大于k时，则当前的i满足题目要求
class Solution {
public:
    vector<int> goodIndices(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> increase(n, 1), decrease(n, 1), ans;
        
        for(int i = 1; i < n; i++) {
            if(nums[i] <= nums[i - 1]) decrease[i] += decrease[i - 1];
        }
        
        for(int i = n - 2; i >= 0; i--) {
            if(nums[i] <= nums[i + 1]) increase[i] += increase[i + 1];
        }
        
        for(int i = k; i < n - k; i++) {
            if(increase[i + 1] >= k && decrease[i - 1] >= k) ans.push_back(i);
        }
        return ans;
    }
};

2423. Remove Letter To Equalize Frequency

踩坑记录：

统计维护最高频率max_freq和次高频率max_freq - 1，满足下列三种情况返回true

  1. 最高频率的字母一共有max_count=1个，意味着剩下的全是max_freq - 1频率的字母
  2. 最高频率为1，其他字母可以直接删除掉
  3. 最低频率的字母有且只有一个
class Solution {
public:
    bool equalFrequency(string word) {
        unordered_map<char, int> m;
        int max_freq = 0, max_count = 0, min_count = 0;
        for(auto c : word) {
            m[c]++;
            max_freq = max(max_freq, m[c]);
        }
        for(auto &[_, freq] : m) {
            if(freq == max_freq) max_count++;
            else if(freq == max_freq - 1) min_count++;
            else return false;
        }
        
        return max_count == 1 || max_freq == 1 || min_count == 1;
    }
};

1996. The Number of Weak Characters in the Game

https://leetcode.com/problems/the-number-of-weak-characters-in-the-game

class Solution {
public:
    int numberOfWeakCharacters(vector<vector<int>>& properties) {
        sort(properties.begin(), properties.end(), [](vector<int> &a, vector<int> &b) {
            if(a[0] != b[0]) return a[0] > b[0];
            else return a[1] < b[1];
        });
        
        int ans = 0, max_def = INT_MIN;
        for(auto p : properties) {
            if(max_def > p[1]) ans++;
            else max_def = p[1];
        }
        return ans;
    }
};

核心思想：

attack属性从大到小排列，defend属性从小到大排列，从而确保当遍历到一个auto p时，当max_def比当前p的defend值大时，attack也能保证比defend大。

2007. Find Original Array From Doubled Array

class Solution {
public:
    vector<int> findOriginalArray(vector<int>& changed) {
        if(changed.size() % 2 == 1) return {};
        unordered_map<int, int> m;
        vector<int> ans;
        for(auto num : changed) {
            m[num]++;
        }
        sort(changed.begin(), changed.end());
        for(auto num : changed) {
            if(m[num] == 0) continue;
            if(m[num * 2] == 0) return {};
            ans.push_back(num);
            m[num]--;
            m[num * 2]--;
        }
        return ans;
    }
};

Notes

  1. 为什么用map而不是set？

    因为要对未来遍历到的元素进行“删除”操作，如果用set会导致越界

  2. 为什么要对changed排序后遍历，而不是用map统计完changed后直接遍历map，如：

    class Solution {
public:
    vector<int> findOriginalArray(vector<int>& changed) {

        map<int, int> m;
        vector<int> ans;
        for(auto num : changed) {
            m[num]++;
        }
        for(auto num : m) {
            if(num.second == 0) continue;
            if(m[num.first * 2] == 0) return {};
            ans.push_back(num.first);
            num.second--;
            m[num.first * 2]--;
        }
        return ans;
    }
};

    因为存在[0, 0, 0, 0]这种特殊情况，expected: [0, 0]，output: [0]

1531. String Compression II

思路分析：

状态转移方程：

dp[i][j]表示第i个字符在删掉j个字符的时候，压缩字符串的长度

如果选择直接删除当前字符，那么dp[i][j] = dp[i + 1][j - 1]

如果不删除当前字符，则从当前往后进行遍历，将与当前字符不同的字符则删除，直到用完删除次数为止：

inline int getcount(int count) {
       return (count == 1 ? 1 : (count < 10 ? 2 : (count < 100 ? 3 : 4)));
}

...

int del_c = j, count = 0;
for(int k = i; k < n && del_c >= 0; k++) {
    if(s[i] == s[k]) {
        count++;
        //这里确保只有当s[i] != s[k + 1]时，才会取到最小值，在取到最小值前的主要目的为更新count
        dp[i][j] = min(dp[i][j], dp[k + 1][del_c] + getcount(count));
    } else {
        del_c--;
    }
}

边界条件：

dp[n][x]时代表一个字符都没选，因此初始长度为0

dp[x][0]表示删了0个字符，因此字符串长度为原长n，但可以用一个随意的最大值替代，如INT_MAX

代码实现：

class Solution {
private:
   int dp[101][101];
   inline int getcount(int count) {
       return (count == 1 ? 1 : (count < 10 ? 2 : (count < 100 ? 3 : 4)));
   }
public:
    int getLengthOfOptimalCompression(string s, int del) {
        int n = s.size();
        for(int i = n; i >= 0; i--) {
            for(int j = 0; j <= del; j++) {
                if(i == n) {
                    dp[i][j] = 0;
                    continue;
                }
                dp[i][j] = (j == 0 ? INT_MAX : dp[i + 1][j - 1]);
                int del_c = j, count = 0;
                for(int k = i; k < n && del_c >= 0; k++) {
                    if(s[i] == s[k]) {
                        count++;
                        dp[i][j] = min(dp[i][j], dp[k + 1][del_c] + getcount(count));
                    } else {
                        del_c--;
                    }
                }
            }
        }
        return dp[0][del];
    }
};

114. Flatten Binary Tree to Linked List

Given the root of a binary tree, flatten the tree into a “linked list”:

  • The “linked list” should use the same TreeNode class where the right child pointer points to the next node in the list and the left child pointer is always null.
  • The “linked list” should be in the same order as a pre-order traversal of the binary tree.

Example 1:

Input: root = [1,2,5,3,4,null,6]
Output: [1,null,2,null,3,null,4,null,5,null,6]

Code:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    TreeNode* dfs(TreeNode* root) {
        if(!root->left && !root->right) return root;
        TreeNode *tail, *next = root->right;
        
        if(root->left != nullptr) {
            tail = dfs(root->left);
            root->right = root->left;
            root->left = nullptr;
            tail->right = next;
        }
        if(root->right != nullptr) tail = dfs(root->right);
        return tail;
    }
public:
    void flatten(TreeNode* root) {
        if(root == nullptr) return;
        dfs(root);
        return;
    }
};

Solution:

Flatten the left subtree, then insert it between the root and the right child, finally go to the right tree and repeat the process before.

1770. Maximum Score from Performing Multiplication Operations

状态转移方程推导：

VERY IMPORTANT! 由于是搜索树，越往下，问题规模越小，下方是上方的子问题，因此要从叶子节点开始构建！
$$
right = n - 1 - (i - left) \
dp[i][left]=\max(dp[i+1][left]+mul[i]*nums[left], dp[i+1][left] + mul[i]*nums[right])
$$
其中，$left$和$right$分别为区间左右侧的下标，$i$为操作次数

代码($O(m^2)$)

class Solution {
public:
    int maximumScore(vector<int>& nums, vector<int>& mul) {
        int n = nums.size(), m = mul.size();
        vector<int> dp(m + 1, 0);
        
        for(int i = m - 1; i >= 0; i--) {
            for(int left = 0; left <= i; left++) {
                int right = n - 1 - (i - left);
                dp[left] = max(dp[left + 1] + mul[i] * nums[left],
                              dp[left] + mul[i] * nums[right]);
            }
        }
        return dp[0];
    }
};

1578. Minimum Time to Make Rope Colorful

踩坑记录：

样例

"bbbaaa" 
[4,9,3,8,8,9]

表明，优先队列必须要用while(q.size() > 1)排空，而不能仅仅只用if(q.size() > 1)

class Solution {
public:
    int minCost(string colors, vector<int>& neededTime) {
        priority_queue<int, vector<int> , greater<int>> q;
        int n = colors.size();
        q.push(neededTime[0]);
        int ans = 0;
        for(int i = 1; i < n; i++) {
            if(colors[i] != colors[i - 1]) {
                while(q.size() > 1) {
                    ans += q.top();
                    q.pop();
                }
                q.pop();
            }
            q.push(neededTime[i]);
        }
        while(q.size() > 1) {
            ans += q.top();
            q.pop();
        }
        return ans;
    }
};

1834. Single-Threaded CPU

思路：

滑动的阈值（time）

踩坑点：

  1. 小顶堆的排序方法必须要使用>，与正常相反
class Solution {
private:
    struct cmp {
        bool operator()(vector<int> const& a, vector<int> const& b) {
            if(a[1] != b[1]) return a[1] > b[1];
            else return a[2] > b[2];
        }
    };
public:
    vector<int> getOrder(vector<vector<int>>& tasks) {
        int n = tasks.size();
        for(int i = 0; i < n; i++) {
            tasks[i].push_back(i);
        }

        sort(tasks.begin(), tasks.end());
        
        priority_queue<vector<int>, vector<vector<int>>, cmp> q;
        vector<int> ans;

        long time = tasks[0][0], i = 1;
        q.push(tasks[0]);

        while(!q.empty()) {
            
            auto p = q.top();
            q.pop();
            time = max(time + p[1], (long)(p[0] + p[1]));
            ans.push_back(p[2]);

            while(i < n && (q.empty() || time >= tasks[i][0])) {
                q.push(tasks[i++]);
            }
        }

        return ans;

    }
};
