729. My Calendar I

You are implementing a program to use as your calendar. We can add a new event if adding the event will not cause a double booking.

A double booking happens when two events have some non-empty intersection (i.e., some moment is common to both events.).

The event can be represented as a pair of integers start and end that represents a booking on the half-open interval [start, end), the range of real numbers x such that start <= x < end.

Implement the MyCalendar class:

  • MyCalendar() Initializes the calendar object.

  • boolean book(int start, int end) Returns true if the event can be added to the calendar successfully without causing a double booking. Otherwise, return false and do not add the event to the calendar.

    class MyCalendar {
    private:
    map<int, int> m;
    public:
    MyCalendar() {

    }

    bool book(int start, int end) {
    auto next = m.upper_bound(start);
    if(next != m.end() && next->second < end) return false;
    else {
    m.insert({end, start});
    return true;
    }
    }
    };

笔记：

  • 二分查找(upper_bound)大于start的end的值，从而避免手写二分查找<=end的算法
  • upper_bound查找end - 1也可以

732. My Calendar III（线段树）

class MyCalendarThree {
private:
    unordered_map<int, int> vals;
    unordered_map<int, int> lazy;

public:
    MyCalendarThree() {}

    void update(int start, int end, int left, int right, int idx) {
        if (start > right || end < left) return;
        if (left >= start && right <= end) {
            lazy[idx]++;
            vals[idx]++;
        } else {
            int mid = (left + right) / 2;
            update(start, end, left, mid, idx * 2);
            update(start, end, mid + 1, right, idx * 2 + 1);
            vals[idx] = lazy[idx] + max(vals[idx * 2], vals[idx * 2 + 1]);
        }
    }

    int book(int start, int end) {
        update(start, end - 1, 1, 1000000000, 1);
        return vals[1];
    }
};

658. Find K Closest Elements

核心思路：

  1. 剪掉k后重新排序，注意题目要求的排序规则
  2. 最后再把k加回来重新排序，满足题目升序要求
class Solution {
public:
    vector<int> findClosestElements(vector<int>& arr, int k, int x) {
        int n = arr.size();
        for(int i = 0; i < n; i++) {
            arr[i] -= x;
        }
        auto cmp = [&](int a, int b) {
            if(a == -b) return a + x < b + x;
            else return abs(a) < abs(b);
        };
        sort(arr.begin(), arr.end(), cmp);
        
        vector<int> ans;
        for(int i = 0; i < k; i++) ans.push_back(arr[i] + x);
        sort(ans.begin(), ans.end());
        return ans;
    }
};

622. Design Circular Queue

教训:

  1. 一定要添加元素个数n
  2. 初始化时，front = 0，rear = -1
class MyCircularQueue {
private:
    int n = 0, k, front = 0, rear = -1;
    vector<int> q;
public:
    MyCircularQueue(int k) {
        this->k = k;
        q.resize(k, 0);
    }
    
    bool enQueue(int value) {
        if(n == k) return false;
        rear = (rear + 1) % k;
        q[rear] = value;
        n++;
        return true;
    }
    
    bool deQueue() {
        if(n == 0) return false;
        
        front = (front + 1) % k;
        n--;
        return true;
    }
    
    int Front() {
        if(n == 0) return -1;
        return q[front];
    }
    
    int Rear() {
        if(n == 0) return -1;
        return q[rear];
    }
    
    bool isEmpty() {
        return n == 0;
    }
    
    bool isFull() {
        return n == k;
    }
};

 * Your MyCircularQueue object will be instantiated and called as such:

 * bool param_2 = obj->deQueue();
 * int param_3 = obj->Front();
 * int param_4 = obj->Rear();
 * bool param_5 = obj->isEmpty();
 * bool param_6 = obj->isFull();

523. Continuous Subarray Sum

数学推导

$(sum[r + 1] - sum[l]) % k = 0$ 变换为$sum[r + 1] % k = sum[l] % k$，因此用hash存储标号，注意长度至少为2

class Solution {
public:
    bool checkSubarraySum(vector<int>& nums, int k) {
        int tot = 0, n = nums.size();
        unordered_map<int, int> m = {{0, 0}};
        for(int i = 0; i < n; i++) {
            tot += nums[i];
            if(!m.count(tot % k)) m[tot % k] = i + 1;
            else if(m[tot % k] < i) return true;
        }
        return false;
    }
};

注意点：

  1. 要初始化为{0, 0}，因为要考虑从头开始全选的情况
  2. 要注意上一个m[tot % k]至少要2长度

336. Palindrome Pairs

前缀数组

由于在本题中，由于需要匹配字符串的前缀，因此我们需要构造一个前缀树，同时在每个TrieNode节点还需要记录子节点是否存在对称的字符串，因此可以构造如下的TireNode

![image-20220917170355236](336. Palindrome Pairs.assets/image-20220917170355236.png “1, 3, 4三种情况”)

struct TrieNode {
        TrieNode* children[26];
        int index = -1;
        vector<int> restIsPal;
};

相比传统的前缀树，多出记录剩下的结点是否为对称节点的vector<int> restIsPal来记录节点标号index

class Solution {
private:
    struct TrieNode {
        TrieNode* children[26];
        int index = -1;
        vector<int> restIsPal;
    };
    
    TrieNode *root = new TrieNode();
    vector<vector<int>> ans;
    
    bool isPal(string &s, int i, int j) {
        while(i < j) {
            if(s[i++] != s[j--]) return false;
        }
        return true;
    }
    void add(string &s, int index) {
        TrieNode *cur = root;
        for(int i = s.size() - 1; i >= 0; i--) {
            int p = s[i] - 'a';
            if(!cur->children[p]) cur->children[p] = new TrieNode();
            if(isPal(s, 0, i)) cur->restIsPal.push_back(index);
            cur = cur->children[p];
        }
        cur->index = index;
    }
    
    void search(string &s, int index) {
        TrieNode* cur = root;
        for(int i = 0; i < s.size(); i++) {
            int p = s[i] - 'a';
            if(cur->index != -1 && isPal(s, i, s.size() - 1)) ans.push_back({index, cur->index});
            if(cur->children[p] == nullptr) return;
            cur = cur->children[p];
        }
        if(cur->index != -1 && index != cur->index) ans.push_back({index, cur->index});
        for(int i : cur->restIsPal) ans.push_back({index, i});
    }
public:
    vector<vector<int>> palindromePairs(vector<string>& words) {
        int n = words.size();
        for(int i = 0; i < n; i++) {
            add(words[i], i);
        }
        
        for(int i = 0; i < n; i++) {
            search(words[i], i);
        }
        return ans;
    }
};

其中add(string s, int index)构建Tire，search(string s, int index)用于搜索当前字符串是否存在一个倒叙相等if(cur->index != -1 && index != cur->index)/倒叙子串if(cur->index != -1 && isPal(s, i, s.size() - 1))，是否为某个字符串的倒叙子串for(int i : cur->restIsPal) ans.push_back({index, i})

Note

  1. 由于我们要处理restIsPal，因此当前i所在的TrieNode并不是i对应的节点，而是其父节点
  2. search时，注意要判子节点是否为空if(cur->children[p] == nullptr) return;

393. UTF-8 Validation

位运算：

class Solution {
public:
    bool validUtf8(vector<int>& data) {
        int n = data.size(), status = 0;
        for(int i = 0; i < n; i++) {
            int mask = 1 << 7, prefix = data[i], count = 0;
            while(mask & prefix) {
                count++;
                mask = mask >> 1;
            }
            if(count > 4) return false;
            if(count > 1) {
                if(status > 0) return false;
                status += count - 1;
            } else if(count == 1) {
                status--;
            }
            
        }
        return status == 0;
    }
};

Notes:

  1. “1”的数量大于4时，直接return false;
  2. count > 1时’status > 0’意味着byte数与“1”的数量不符，return false;
  3. 保证最后加减相抵消，即可保证算法正确性

290. Word Pattern

踩坑点：

  1. c++ 下如何split：

​ 对于 " "来说，可以使用sstream来进行分割

  1. 相互映射，不止string->char，还得char->string

源代码：

class Solution {
public:
    bool wordPattern(string pattern, string s) {
        stringstream ss(s);
        istream_iterator<string> begin(ss), end;
        vector<string> s_arr(begin, end);
        if(pattern.size() != s_arr.size()) return false;
        int n = pattern.size();
        unordered_map<char, string> m;
        unordered_map<string, char> h;
        for(int i = 0; i < n; i++) {
            if(m.count(pattern[i]) == 0 && h.count(s_arr[i]) == 0) {
                m[pattern[i]] = s_arr[i];
                h[s_arr[i]] = pattern[i];
            } else if(m[pattern[i]] != s_arr[i] || h[s_arr[i]] != pattern[i]) {
                return false;
            }
        }
        return  true;
    }
};

732. My Calendar III

核心思想：扫描线段

用一个map记录多个线段的起点和终点，起点++，终点–，最后从小到大扫描map，记录最大的重叠部分

class MyCalendarThree {
private:
    map<int, int> m;
public:
    MyCalendarThree() {
        
    }
    
    int book(int start, int end) {
        m[start]++;
        m[end]--;
        int ans = 0, lad = 0;
        for(auto &[_, num] : m) {
            lad += num;
            ans = max(ans, lad);
        }
        return ans;
    }
};

218. The Skyline Problem

踩坑记录

  1. 要处理单横坐标点上存在多个建筑的情况
  2. 用于存储的二维数组，横坐标要从小到大排序，而高度要从大到小排序才能保证天际线不会因为单坐标点上出现多个建筑导致结果“冗余”
  3. 由于单坐标点存在多个建筑，因此不能再维护一个单调数组（’c++中可用set实现）了，而是应该使用map来记录每个横坐标点上的建筑数量

代码：

class Solution {
public:
    vector<vector<int>> getSkyline(vector<vector<int>>& buildings) {
        vector<pair<int, int>> key_points;
        
        for(auto building : buildings) {
            int left = building[0], right = building[1], height = building[2];
            key_points.push_back({left, height});
            key_points.push_back({right, -height});
        }
        sort(key_points.begin(), key_points.end(), [](auto a, auto b) {
            if(a.first == b.first) return a.second > b.second;
            else return a.first < b.first;
        });
        map<int, int, greater<int>> points;
        vector<vector<int>> ans;
        int max_height = 0;
        for(auto &[index, height] : key_points) {
            if(height > 0) points[height]++;
            else points[-height]--;
            if(points[abs(height)] == 0) points.erase(abs(height));
            int top = points.begin()->first;
            if(top != max_height)ans.push_back({index, top});
            max_height = top;
        }
        return ans;
    }
};
