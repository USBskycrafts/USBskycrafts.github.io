1043. Partition Array for Maximum Sum

DP：

class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& arr, int k) {
        int n = arr.size();
        int dp[n + 1];
        memset(dp, 0, sizeof(dp));
        for(int i = n - 1; i >= 0; i--) {
            int len = 1, max_num = 0, max_sum = 0;
            for(int j = i; j < min(n, i + k); j++) {
                max_num = max(max_num, arr[j]);
                max_sum = max((len++) * max_num + dp[j + 1], max_sum);
            }
            dp[i] = max_sum;
        }
        return dp[0];
    }
};

Note:

  1. 先后缀，再前缀的扫描，可以充分对结果max_num进行记录

948. Bag of Tokens

双指针（$O(n\log{n}))$）

class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int power) {
        sort(tokens.rbegin(), tokens.rend());
        int i = 0, j = tokens.size() - 1;
        int score = 0;
        while(i <= j) {
            if(power < tokens[j] && score == 0) break;
            
            if(power < tokens[j]) {
                power += (tokens[i] - tokens[j]);
                i++, j--;
            } else {
                score += 1;
                power -= tokens[j];
                j--;
            }
            //cout << power << endl;
        }
        return score;
    }
};

Notes:

  1. 必须判断当前score能否足够转换成powerif(power < tokens[j] && score == 0) break;，然后才能进行token到power的转换

    if(power < tokens[j]) {
    power += (tokens[i] - tokens[j]);
    i++, j--;
}

  2. 由于是从大到小进行了排序，因此if(power < tokens[j])里token[i]一定足够转换成token[j]

838. Push Dominoes

Core: quantitative the force

if ‘L’ is ‘R’ is in initial string, we give the highest force, then when it spread, it is count down at the adjacent num, we do it from left and from right

class Solution {
public:
    string pushDominoes(string dominoes) {
        int n = dominoes.size();
        auto print = [&](int a){cout << a << " ";};
        vector<int> left(n, 0), right(n, 0);
        string ans(n, '.');
        for(int i = 0; i < n; i++) {
            if(dominoes[i] == 'R') left[i] = n;
            else if(dominoes[i] == 'L') continue;
            else if(i != 0 && left[i - 1] > 0) left[i] = left[i - 1] - 1;
            
        }
        
        for(int i = n - 1; i >= 0; i--) {
            if(dominoes[i] == 'L') right[i] = n;
            else if(dominoes[i] == 'R') continue;
            else if(i != n - 1 && right[i + 1] > 0) right[i] = right[i + 1] - 1;
        }
        
        for(int i = 0; i < n; i++) {
            if(left[i] > right[i]) ans[i] = 'R';
            else if(left[i] < right[i]) ans[i] = 'L';
            else ans[i] = '.';
        }
        
        for_each(left.begin(), left.end(), print);
        cout << endl;
        for_each(right.begin(), right.end(), print);
        cout << endl;
        
        return ans;
    }
};

834. Sum of Distances in Tree

Analysis:

Constraints:

  • 1 <= n <= 3 * 104
  • edges.length == n - 1
  • edges[i].length == 2
  • 0 <= ai, bi < n
  • ai != bi
  • The given input represents a valid tree.

Because of that, the algorithm must be $O(n)$, so must analyze the relation between each nodes

#Y represent the node number of Y subtree.

anx[y] = x@X + y@Y + #(X), so ans[x] - ans[y] = #(Y) - #(X)

So, first we search from root to leaves to get the distance from root to every node, then we use the equation above to iterate to get the answer, like the picture below

class Solution {
private:
    unordered_map<int, vector<int>> graph;
    vector<int> count, dis, ans;
    
    inline void dfs1(int root = 0, int parent = -1) {        
        for(auto node : graph[root]) {
            if(node == parent) continue;
            dfs1(node, root);
            count[root] += count[node];
            dis[root] += dis[node] + count[node];
        }
    }
    
    inline void dfs2(int root = 0, int parent = -1) {
        for(auto node : graph[root]) {
            if(node == parent) continue;
            int n = count.size();
            ans[node] = ans[root] + (n - count[node]) - count[node];
            dfs2(node, root);
        }
    }
public:
    vector<int> sumOfDistancesInTree(int n, vector<vector<int>>& edges) {
        count.resize(n, 1), dis.resize(n, 0), ans.resize(n, 0);
        int m = edges.size();
        for(int i = 0; i < m; i++) {
            int u = edges[i][0], v = edges[i][1];
            graph[u].push_back(v);
            graph[v].push_back(u);
        }
        dfs1();
        ans[0] = dis[0];
        dfs2();
        return ans;
    }
};

class Solution {
private:
    int mod = (int)(1e9 + 7);
public:
    int numTilings(int n) {
        vector<long> T(n + 1, 0), D(n + 1, 0);
        D[0] = 1;
        if(n >= 1) {
            D[1] = 1;
        } 
        if(n >= 2) {
            D[2] = 2;
            T[2] = 2;
        }

        for(int i = 3; i <= n; i++) {
            D[i] += ((D[i - 2] + D[i - 1]) % mod  + T[i - 1]) % mod;
            T[i] += ((D[i - 2] * 2) % mod + T[i - 1]) % mod;
        }

        return D[n] % (int)(1e9 + 7);
    }
};

767. Reorganize String

思路：

取堆上数量最多的两个字母，然后交错排序。因为是交错排序，因此，要分别用cur，prev的结构

class Solution {
public:
    string reorganizeString(string s) {
        unordered_map<char, int> m;
        for(char c : s) m[c]++;
        priority_queue<pair<int, char>> h;
        for(auto p : m) h.push({p.second, p.first});
        pair<int, char> prev = {0, '#'}, cur;
        string ans;
        while(h.size()) {
            cur = h.top();
            h.pop();
            ans += cur.second;
            cur.first--;
            if(prev.first > 0) h.push(prev);
            prev = cur;
        }
        cout << ans << endl;
        return ans.size() == s.size() ? ans : "";
    }
};

729. My Calendar I

You are implementing a program to use as your calendar. We can add a new event if adding the event will not cause a double booking.

A double booking happens when two events have some non-empty intersection (i.e., some moment is common to both events.).

The event can be represented as a pair of integers start and end that represents a booking on the half-open interval [start, end), the range of real numbers x such that start <= x < end.

Implement the MyCalendar class:

  • MyCalendar() Initializes the calendar object.

  • boolean book(int start, int end) Returns true if the event can be added to the calendar successfully without causing a double booking. Otherwise, return false and do not add the event to the calendar.

    class MyCalendar {
    private:
    map<int, int> m;
    public:
    MyCalendar() {

    }

    bool book(int start, int end) {
    auto next = m.upper_bound(start);
    if(next != m.end() && next->second < end) return false;
    else {
    m.insert({end, start});
    return true;
    }
    }
    };

笔记：

  • 二分查找(upper_bound)大于start的end的值，从而避免手写二分查找<=end的算法
  • upper_bound查找end - 1也可以

732. My Calendar III（线段树）

class MyCalendarThree {
private:
    unordered_map<int, int> vals;
    unordered_map<int, int> lazy;

public:
    MyCalendarThree() {}

    void update(int start, int end, int left, int right, int idx) {
        if (start > right || end < left) return;
        if (left >= start && right <= end) {
            lazy[idx]++;
            vals[idx]++;
        } else {
            int mid = (left + right) / 2;
            update(start, end, left, mid, idx * 2);
            update(start, end, mid + 1, right, idx * 2 + 1);
            vals[idx] = lazy[idx] + max(vals[idx * 2], vals[idx * 2 + 1]);
        }
    }

    int book(int start, int end) {
        update(start, end - 1, 1, 1000000000, 1);
        return vals[1];
    }
};

658. Find K Closest Elements

核心思路：

  1. 剪掉k后重新排序，注意题目要求的排序规则
  2. 最后再把k加回来重新排序，满足题目升序要求
class Solution {
public:
    vector<int> findClosestElements(vector<int>& arr, int k, int x) {
        int n = arr.size();
        for(int i = 0; i < n; i++) {
            arr[i] -= x;
        }
        auto cmp = [&](int a, int b) {
            if(a == -b) return a + x < b + x;
            else return abs(a) < abs(b);
        };
        sort(arr.begin(), arr.end(), cmp);
        
        vector<int> ans;
        for(int i = 0; i < k; i++) ans.push_back(arr[i] + x);
        sort(ans.begin(), ans.end());
        return ans;
    }
};

622. Design Circular Queue

教训:

  1. 一定要添加元素个数n
  2. 初始化时，front = 0，rear = -1
class MyCircularQueue {
private:
    int n = 0, k, front = 0, rear = -1;
    vector<int> q;
public:
    MyCircularQueue(int k) {
        this->k = k;
        q.resize(k, 0);
    }
    
    bool enQueue(int value) {
        if(n == k) return false;
        rear = (rear + 1) % k;
        q[rear] = value;
        n++;
        return true;
    }
    
    bool deQueue() {
        if(n == 0) return false;
        
        front = (front + 1) % k;
        n--;
        return true;
    }
    
    int Front() {
        if(n == 0) return -1;
        return q[front];
    }
    
    int Rear() {
        if(n == 0) return -1;
        return q[rear];
    }
    
    bool isEmpty() {
        return n == 0;
    }
    
    bool isFull() {
        return n == k;
    }
};
